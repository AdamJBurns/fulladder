CHIP FullAdder{
    IN a, b, c;
    OUT sum, carry;

    PARTS:
    HalfAdder(a=a, b=b, sum=sumgate1, carry=carrygate1);
    HalfAdder(a=sumgate1, b=c, sum=sum, carry=carrygate2);
    Or(a=carrygate1, b=carrygate2, out=carry);

}
